`SELECT`
    `e.first_name,`

Para hacer la salida más clara y comprensible, podemos reemplazar los valores NULL por un mensaje descriptivo que explique en el reporte qué significa la ausencia de ese dato.

`SELECT`
    `e.first_name,`
    `e.last_name,`
    `j.job_title,`
     `NVL(d.department_name, 'Sin departamento') AS departamento`
`FROM     employees   e`
`JOIN     jobs        j  ON e.job_id        = j.job_id`
`LEFT JOIN departments d ON e.department_id = d.department_id`
`ORDER BY d.department_name , e.last_name;`

```
SELECT 
    co.country_name,
    d.department_name,
    COUNT(e.employee_id) AS cantidad_empleados
FROM employees e
JOIN departments d  ON e.department_id = d.department_id
JOIN locations   l  ON d.location_id   = l.location_id
JOIN countries   co ON l.country_id    = co.country_id
GROUP BY co.country_name, d.department_name
ORDER BY co.country_name, cantidad_empleados DESC;
```

```
SELECT NVL(c.country_name,'Sin pais'), d.department_name, count(e.employee_id),
CASE WHEN
COUNT(e.employee_id)>5 THEN 'Muchos Empleados'
ELSE 'Pocos Empleados'
END
FROM employees e LEFT JOIN departments d ON e.department_id=d.department_id
LEFT JOIN locations l ON l.location_id=d.location_id
LEFT JOIN countries c ON l.country_id=c.country_id
GROUP BY c.country_name,d.department_name
HAVING c.country_name not like 'Canada'
```
